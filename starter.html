def save_images(image_dataframe, image_data, save_directory):
    # Extract rows and filenames from the dataframe
    row_indices = image_dataframe["row"]
    image_filenames = image_dataframe["filenames"]
    
    # Loop through each row index and corresponding filename
    for row_index, filename in zip(row_indices, image_filenames):
        # Obtain the image data for the current row
        image = image_data[b'data'][row_index]

        # Reshape and stack the image data for each color channel
        reshaped_image = np.dstack((
            image[0:1024].reshape(32, 32, 1),
            image[1024:2048].reshape(32, 32, 1),
            image[2048:].reshape(32, 32, 1)
        ))

        save_path = os.path.join(save_directory, filename)
        plt.imsave(save_path, reshaped_image)
        print(f"Image {filename} saved.")

For Image Classification, Sagemaker also expects metadata e.g. in the form of TSV files with labels and filepaths. We can generate these using our Pandas DataFrames from earlier:

In[34]:
def to_metadata_file(df, prefix):
    df["s3_path"] = df["filenames"]
    df["labels"] = df["labels"].apply(lambda x: 0 if x==8 else 1)
    return df[["row", "labels", "s3_path"]].to_csv(
        f"{prefix}.lst", sep="\t", index=False, header=False
    )
    
to_metadata_file(df_train.copy(), "train")
to_metadata_file(df_test.copy(), "test")

from sagemaker.model_monitor import DataCaptureConfig
